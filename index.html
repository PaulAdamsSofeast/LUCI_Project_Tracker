function today() { const d = new Date(); d.setHours(0,0,0,0); return d; }


function fmtDate(d) {

  if (!d) return '';

  const dt = (d instanceof Date) ? d : new Date(d);

  if (isNaN(dt)) return '';

  return `${dt.getFullYear()}-${String(dt.getMonth()+1).padStart(2,'0')}-${String(dt.getDate()).padStart(2,'0')}`;

}


function fmtDateDisplay(d) {

  if (!d) return '';

  const dt = (d instanceof Date) ? d : new Date(d + 'T00:00:00');

  if (isNaN(dt)) return '';

  return dt.toLocaleDateString('en-GB', { day:'2-digit', month:'short', year:'numeric' });

}


function parseDate(s) {

  if (!s) return null;

  const d = new Date(s + 'T00:00:00');

  if (isNaN(d)) return null;

  d.setHours(0,0,0,0);

  return d;

}


function addDays(d, n) { const r = new Date(d); r.setDate(r.getDate()+n); return r; }

function daysBetween(a, b) { return Math.round((b-a)/(1000*60*60*24)); }

function isWeekend(d) { return d.getDay()===0 || d.getDay()===6; }


function escapeHtml(s) {

  if (s==null) return '';

  return String(s).replace(/[&<>"']/g, c => ({'&':'&amp;','<':'&lt;','>':'&gt;','"':'&quot;',"'":'&#39;'}[c]));

}


function resolveEffectiveStart(task, visited = new Set()) {

  if (visited.has(task.id)) return parseDate(task.start);

  if (!task.deps || !task.deps.trim()) return parseDate(task.start);

  visited.add(task.id);

  const depIds = task.deps.split(/[,\s]+/).map(s=>s.trim()).filter(Boolean);

  let maxDepEnd = null;

  for (const depId of depIds) {

    const dep = tasks.find(t => t.id.toLowerCase() === depId.toLowerCase());

    if (!dep) continue;

    const depStart = resolveEffectiveStart(dep, new Set(visited));

    if (!depStart) continue;

    const depEnd = addDays(depStart, Math.max(1, dep.duration||1));

    if (!maxDepEnd || depEnd > maxDepEnd) maxDepEnd = depEnd;

  }

  const manual = parseDate(task.start);

  if (maxDepEnd) return (manual && manual > maxDepEnd) ? manual : maxDepEnd;

  return manual;

}


function isDepDriven(task) {

  if (!task.deps || !task.deps.trim()) return false;

  for (const depId of task.deps.split(/[,\s]+/).map(s=>s.trim()).filter(Boolean)) {

    const dep = tasks.find(t => t.id.toLowerCase() === depId.toLowerCase());

    if (!dep) continue;

    const depEnd = addDays(resolveEffectiveStart(dep, new Set([task.id])), Math.max(1, dep.duration||1));

    const manual = parseDate(task.start);

    if (!manual || depEnd >= manual) return true;

  }

  return false;

}

function onDragOver(e, idx) {

  e.preventDefault();

  if (dragSrcIdx === null || dragSrcIdx === idx) return;

  const rect = tr.getBoundingClientRect();

  document.querySelectorAll('#task-body tr').forEach(r => r.classList.remove('drag-above','drag-below'));

function getTaskDateRange() {

  const starts = tasks.map(t => resolveEffectiveStart(t)).filter(Boolean);

  if (!starts.length) return { start: addDays(today(),-3), end: addDays(today(),42) };

  const minS = new Date(Math.min(...starts.map(d=>d.getTime())));

  const maxE = new Date(Math.max(...tasks.map(t => {

    const s = resolveEffectiveStart(t);

    return s ? addDays(s, t.duration||1).getTime() : 0;

  })));

  return { start: addDays(minS,-2), end: addDays(maxE,4) };

}

document.addEventListener('focusin', e => {

  if (e.target.closest('.task-wrap, .milestones-bar, .info-bar')) focusedInEditor = true;

});


document.addEventListener('focusout', () => {

  setTimeout(() => {

    focusedInEditor = !!document.querySelector('.task-wrap :focus, .milestones-bar :focus, .info-bar :focus');

    if (!focusedInEditor && pendingRemote) {

      clearTimeout(pendingRemoteTimer);

      applyRemoteState(pendingRemote);

      pendingRemote = null;

      const notice = document.getElementById('pending-update-notice');

      if (notice) notice.style.display = 'none';

    }

  }, 200);

});


function syncMeta() { pushToFirebase(); }


function pushToFirebase() {

  if (!db || !projectId) return;

  clearTimeout(syncTimer);

  setSyncStatus('syncing');

  syncTimer = setTimeout(() => {

    const payload = {

      meta: {

        name:   document.getElementById('proj-name').value,

        client: document.getElementById('proj-client').value,

        pm:     document.getElementById('proj-pm').value,

        status: document.getElementById('proj-status').value,

      },

      tasks, milestones, nextId, nextMsId,

      version, versionDate,

      _writer: clientId,

      _ts:     firebase.database.ServerValue.TIMESTAMP,

    };

    db.ref(`/agilian/${projectId}`).set(payload)

      .then(()   => setSyncStatus('synced'))

      .catch(err => { console.error('Firebase write error', err); setSyncStatus('error'); });

  }, 300);

}


function handleRemoteSnapshot(snapshot) {

  const data = snapshot.val();

  if (!data) return;

  if (data._writer === clientId) return;

  if (focusedInEditor) {

    pendingRemote = data;

    const notice = document.getElementById('pending-update-notice');

    if (notice) notice.style.display = 'inline';

    clearTimeout(pendingRemoteTimer);

    pendingRemoteTimer = setTimeout(() => {

      if (pendingRemote) {

        applyRemoteState(pendingRemote);

        pendingRemote = null;

        const n = document.getElementById('pending-update-notice');

        if (n) n.style.display = 'none';

      }

    }, 8000);

  } else {

    applyRemoteState(data);

  }

}


function applyRemoteState(data) {

  if (data.meta) {

    document.getElementById('proj-name').value   = data.meta.name   || '';

    document.getElementById('proj-client').value = data.meta.client || '';

    document.getElementById('proj-pm').value     = data.meta.pm     || '';

    document.getElementById('proj-status').value = data.meta.status || '';

  }

  if (Array.isArray(data.tasks)) {

    tasks  = data.tasks;

    nextId = data.nextId || (Math.max(0, ...tasks.map(t=>parseInt(t.id.slice(1))||0)) + 1);

  }

  if (Array.isArray(data.milestones)) {

    milestones = data.milestones;

    nextMsId   = data.nextMsId || (milestones.length + 1);

  }

  if (data.version != null) {

    version     = data.version;

    versionDate = data.versionDate || '';

    updateVersionDisplay();

  }

  render();

}

}

function processFile(file) {

  if (!file.name.match(/\.(xlsx|xls|csv)$/i)) { alert('Please select an Excel (.xlsx/.xls) or CSV file.'); return; }

  document.getElementById('drop-zone').querySelector('.dz-title').textContent = file.name;

  document.getElementById('drop-zone').querySelector('.dz-sub').textContent = (file.size/1024).toFixed(1)+' KB — loading…';

  const reader = new FileReader();

  reader.onload = e => {

    try {

      const data = new Uint8Array(e.target.result);

      importWorkbook = XLSX.read(data, { type:'array', cellDates:true });

      const sel = document.getElementById('sheet-select');

      sel.innerHTML = importWorkbook.SheetNames.map(n=>`<option value="${escapeHtml(n)}">${escapeHtml(n)}</option>`).join('');

      document.getElementById('drop-zone').querySelector('.dz-sub').textContent = `${importWorkbook.SheetNames.length} sheet(s) found`;

      document.getElementById('sheet-picker').style.display = 'block';

      loadSheet();

    } catch (err) { alert('Could not read file: '+err.message); }

  };

  reader.readAsArrayBuffer(file);

}


function loadSheet() {

  const ws = importWorkbook.Sheets[document.getElementById('sheet-select').value];

  if (!ws) return;

  const raw = XLSX.utils.sheet_to_json(ws, { header:1, defval:'' });

  if (raw.length < 2) { alert('Sheet appears empty or has only one row.'); return; }

  importHeaders = raw[0].map(h=>String(h).trim());

  importRows    = raw.slice(1).filter(r=>r.some(c=>c!==''));

  populateMappingDropdowns();

  autoDetectMappings();

  document.getElementById('mapping-section').style.display = 'block';

  updatePreview();

}


function populateMappingDropdowns() {

  const opts = '<option value="">— skip —</option>' + importHeaders.map((h,i)=>`<option value="${i}">${escapeHtml(h)}</option>`).join('');

  ['map-name','map-owner','map-start','map-duration','map-deps','map-status','map-progress'].forEach(id => { document.getElementById(id).innerHTML = opts; });

}


function autoDetectMappings() {

  const p = {

    'map-name':     /task|name|title|description|activity/i,

    'map-owner':    /owner|assigned|resource|who|responsible/i,

    'map-start':    /start|begin|commence|from/i,

    'map-duration': /duration|days|length|effort/i,

    'map-deps':     /depend|predecessor|after|requires/i,

    'map-status':   /status|state|phase/i,

    'map-progress': /progress|complete|done|percent|%/i,

  };

  Object.entries(p).forEach(([id, rx]) => {

    const m = importHeaders.findIndex(h=>rx.test(h));

    if (m !== -1) document.getElementById(id).value = String(m);

  });

  ['map-name','map-owner','map-start','map-duration','map-deps','map-status','map-progress']

    .forEach(id => { document.getElementById(id).onchange = updatePreview; });

}


function getColVal(row, selectId) {

  const sel=document.getElementById(selectId); if (!sel||sel.value==='') return '';

  const v=row[parseInt(sel.value,10)]; return (v===undefined||v===null)?'':String(v).trim();

}


function parseExcelDate(raw) {

  if (!raw||raw==='') return '';

  if (raw instanceof Date && !isNaN(raw)) return fmtDate(raw);

  if (typeof raw==='number') { const d=XLSX.SSF.parse_date_code(raw); if(d) return fmtDate(new Date(d.y,d.m-1,d.d)); }

  const a=new Date(String(raw)); return isNaN(a)?'':fmtDate(a);

}


function normaliseStatus(raw) {

  const s=String(raw).toLowerCase();

  if (/complete|done|finish|closed/i.test(s))    return 'Complete';

  if (/progress|active|ongoing|started/i.test(s)) return 'In progress';

  if (/risk|concern|warn/i.test(s))               return 'At risk';

  if (/block|hold|stuck|wait/i.test(s))           return 'Blocked';

  return 'Not started';

}


function buildMappedTasks() {

  return importRows.map(row => {

    const name=getColVal(row,'map-name'), owner=getColVal(row,'map-owner');

    const rs=getColVal(row,'map-start'), dur=parseInt(getColVal(row,'map-duration'),10);

    const deps=getColVal(row,'map-deps'), status=getColVal(row,'map-status');

    const progress=parseInt(getColVal(row,'map-progress'),10);

    return {

      name, owner,

      start:    parseExcelDate(rs)||rs,

      duration: isNaN(dur)||dur<1?5:dur,

      deps,

      status:   status?normaliseStatus(status):'Not started',

      progress: isNaN(progress)?0:Math.max(0,Math.min(100,progress)),

    };

  }).filter(t=>t.name);

}


function updatePreview() {

  const mapped=buildMappedTasks();

  document.getElementById('preview-section').style.display = mapped.length?'block':'none';

  document.getElementById('import-btn').disabled = !mapped.length;

  document.getElementById('preview-count').textContent = `(${mapped.length} task${mapped.length!==1?'s':''} ready to import)`;

  const cols=['name','owner','start','duration','deps','status','progress'];

  const heads=['Task','Owner','Start','Days','Depends On','Status','% Done'];

  const preview=mapped.slice(0,8);

  document.getElementById('preview-table').innerHTML =

    `<thead><tr>${heads.map(h=>`<th>${h}</th>`).join('')}</tr></thead>

     <tbody>${preview.map(t=>`<tr>${cols.map(c=>`<td title="${escapeHtml(String(t[c]))}">${escapeHtml(String(t[c]??''))}</td>`).join('')}</tr>`).join('')

     }${mapped.length>8?`<tr><td colspan="7" style="text-align:center;color:var(--muted);padding:8px;font-style:italic;">… and ${mapped.length-8} more rows</td></tr>`:''}</tbody>`;

}

function saveFile() {

  const data = {

    project: {

      name:   document.getElementById('proj-name').value,

      client: document.getElementById('proj-client').value,

      pm:     document.getElementById('proj-pm').value,

      status: document.getElementById('proj-status').value,

    },

    tasks, milestones, nextId, nextMsId, version, versionDate

  };

  Object.entries(data.project).forEach(([k, v]) => {

    const el = document.getElementById('proj-' + k);

    if (el) el.setAttribute('value', v);

  });

  let html = document.documentElement.outerHTML;

  html = html.replace(/<script[^>]+id="saved-state"[^>]*>[\s\S]*?<\/script>\s*/g, '');

  const script   = `<script id="saved-state" type="application/json">${JSON.stringify(data).replace(/<\//g,'<\\/')}<\/script>`;

  const injected = html.replace('</head>', script + '\n</head>');

  const blob = new Blob(['<!DOCTYPE html>\n' + injected], { type: 'text/html' });

  const url  = URL.createObjectURL(blob);

  const a    = document.createElement('a');

  a.href     = url;

  a.download = `${(data.project.name||'project').replace(/[^a-z0-9]+/gi,'_').toLowerCase()}_tracker_${fmtDate(new Date())}.html`;

  document.body.appendChild(a); a.click(); document.body.removeChild(a);

  URL.revokeObjectURL(url);

}

    if (data.version != null) { version = data.version; versionDate = data.versionDate || ''; }

    return true;

  } catch(e) { console.error(e); return false; }

}
